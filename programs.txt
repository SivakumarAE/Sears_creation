		int longwordlength=0;

		for (int i = 0; i < sentance.length(); i++) {
		char ch = sentance.charAt(i);
		if (ch!=' ') {
			word = word+ch;
		}
		else{
			wordlength=word.length();
			if(longwordlength<=wordlength) {
				longword=word;
				longwordlength=wordlength;
			}

				smallwordlength=wordlength;
				smallword=word;
			}
			word="";
			
		}	
		}
